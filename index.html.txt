    <script>
        // Array para almacenar los items del carrito
        let cart = [];

        // Función para añadir un item al carrito
        function addToCart(item) {
            const existingItem = cart.find(cartItem => cartItem.id === item.id);
            if (existingItem) {
                existingItem.quantity += 1;
            } else {
                cart.push({ ...item, quantity: 1 });
            }
            updateCartCount();
            showAlert();
        }

        // Función para actualizar el contador del carrito en el botón
        function updateCartCount() {
            const totalItems = cart.reduce((total, item) => total + item.quantity, 0);
            document.getElementById('cartCount').textContent = totalItems;
        }

        // Función para mostrar una alerta de éxito temporal
        function showAlert() {
            const alertBox = document.getElementById('alert');
            alertBox.classList.remove('hidden');
            setTimeout(() => {
                alertBox.classList.add('hidden');
            }, 3000);
        }

        // Función para mostrar el modal del carrito
        function showCart() {
            const cartModal = document.getElementById('cartModal');
            const cartItemsContainer = document.getElementById('cartItems');
            const cartTotalElement = document.getElementById('cartTotal');
            let total = 0;
            
            cartItemsContainer.innerHTML = '';
            
            if (cart.length === 0) {
                cartItemsContainer.innerHTML = '<p class="text-center text-gray-500">Tu carrito está vacío.</p>';
            } else {
                cart.forEach(item => {
                    const itemTotal = item.price * item.quantity;
                    total += itemTotal;
                    const itemElement = document.createElement('div');
                    itemElement.className = 'flex items-center justify-between bg-white p-3 rounded-lg shadow-sm';
                    itemElement.innerHTML = `
                        <div>
                            <span class="font-semibold text-gray-800">${item.name}</span>
                            <span class="text-sm text-gray-500">(${item.quantity}x)</span>
                        </div>
                        <span class="font-bold text-green-600">$${itemTotal}</span>
                    `;
                    cartItemsContainer.appendChild(itemElement);
                });
            }
            
            cartTotalElement.textContent = `$${total}`;
            cartModal.style.display = 'flex';
        }

        // Función para mostrar el modal de checkout
        function showCheckout() {
            if (cart.length === 0) {
                const checkoutError = document.getElementById('checkoutError');
                checkoutError.textContent = 'Tu carrito está vacío. Agrega productos primero.';
                checkoutError.classList.remove('hidden');
                return;
            }

            const checkoutModal = document.getElementById('checkoutModal');
            const checkoutTotalElement = document.getElementById('checkoutTotal');
            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
            
            checkoutTotalElement.textContent = `$${total}`;
            closeModal('cartModal');
            checkoutModal.style.display = 'flex';
        }

        // Función para vaciar el carrito
        function clearCart() {
            cart = [];
            updateCartCount();
            closeModal('cartModal');
        }

        // Función para cerrar un modal
        function closeModal(modalId) {
            document.getElementById(modalId).style.display = 'none';
        }

        // Función para enviar el pedido por WhatsApp
        function sendOrderViaWhatsApp() {
            const nameInput = document.getElementById('nameInput').value.trim();
            const locationInput = document.getElementById('locationInput').value.trim();
            const paymentInput = document.getElementById('paymentInput').value;
            const checkoutError = document.getElementById('checkoutError');
            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);

            if (!nameInput) {
                checkoutError.textContent = 'Por favor, ingresa tu nombre.';
                checkoutError.classList.remove('hidden');
                return;
            }

            if (!locationInput) {
                checkoutError.textContent = 'Por favor, ingresa tu ubicación.';
                checkoutError.classList.remove('hidden');
                return;
            }

            const paymentAmount = parseFloat(paymentInput);
            if (isNaN(paymentAmount) || paymentAmount < total) {
                checkoutError.textContent = 'El monto de pago debe ser mayor o igual al total.';
                checkoutError.classList.remove('hidden');
                return;
            }
            
            checkoutError.classList.add('hidden');
            const change = paymentAmount - total;

            // Construye el mensaje para WhatsApp
            let message = "🎉 *¡Nuevo Pedido!* 🎉\n\n";
            message += `*Cliente:* ${nameInput}\n`;
            message += "*Detalles del pedido:*\n";
            cart.forEach(item => {
                message += `${item.name} (${item.quantity}x) - $${item.price * item.quantity}\n`;
            });
            message += "\n--------------------\n";
            message += `*Total:* $${total}\n`;
            message += `*Pago con:* $${paymentAmount}\n`;
            message += `*Cambio a entregar:* $${change}\n`;
            message += `*Ubicación del cliente:* ${locationInput}\n`;
            message += "\n¡A preparar el pedido! 🍿";

            const phoneNumber = "18496322671";

            const encodedMessage = encodeURIComponent(message);
            const whatsappUrl = `https://api.whatsapp.com/send?phone=${phoneNumber}&text=${encodedMessage}`;

            window.open(whatsappUrl, '_blank');

            // Limpia el carrito y cierra el modal después de enviar
            cart = [];
            updateCartCount();
            closeModal('checkoutModal');
        }
    </script>
